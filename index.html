Get the Middle Character
7 kuy kata from Codewars
You are going to be given a word. Your job is to return the middle character of the word. If the word's length is odd, return the middle character. If the word's length is even, return the middle 2 characters. 

                        
    function getMiddle(s)
    {
        if(s.length % 2 == 0) {
        let firstHalfOfString = s.slice(0, s.length / 2);
        firstLetter = firstHalfOfString.slice(-1);
    
        let secondHalfOfString = s.slice(s.length / 2);
        secondLetter = secondHalfOfString.slice(0,1);
    
        return firstLetter + secondLetter;
        
        } else if(s.length % 2 != 0) {
            let startOfString = s.slice(0, s.length / 2 + 0.5);
        
            return startOfString.slice(-1);
        }
    }
